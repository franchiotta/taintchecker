752/* a and b must be the same size, which is n2.
753 * r needs to be n2 words and t needs to be n2*2

757void bn_mul_high(BN_ULONGunsigned long *r, BN_ULONGunsigned long *a, BN_ULONGunsigned long *b, BN_ULONGunsigned long *l, int n2,

771 neg=zero=0;
772 c1=bn_cmp_words(&(a[0]),&(a[n]),n);

809 /* r[10] = (a[1]*b[1]) */
810# ifdef BN_MUL_COMBA

828 */

906 } while (lc);
907 }
908 else

937 } while (lc);

939 }
